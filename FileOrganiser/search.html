# Search the dataframe of folder/files and return exact name matches - either matches whole word or phrase 
def ExactNameSearch(name:str, tree:Tree, ignoreCase=True, phrase=False):
    """
        Description:
        ------------
        Search the dataframe names column for exact matches. 
        If phrase=True this matches any string that contains the passed name. 
        Otherwise only full words are matched if phrase=False.


        Args:
        -----  
            name : str 
                #: The name to search for 
            tree : Tree
                #: The tree class containing the dataframes
        Kwargs:
        -------
            ignoreCase : bool : default = True
                #: Case sensitive/insensitive search
            phrase : bool : default = True
                #: Match whole word only or any string containing phrase

        Outputs:
        --------
            output : tuple(list, list) 
                #: Tuple containing the list of paths (df index) of the found search items 
                in the following form: (folderResults, fileResults)

        Notes:
        ------
        The Search_MultipleNamesInclusive method produces results that (inclusively) match
        against a list of names using repeated application of this method for each name.

    """
    if(not phrase):
        # Regex for matching whole word only
        regex = r"(\b" + name + r"\b)"
    if(phrase):
        # Regex for matching phrase
        regex = r"(" + name + r")"


    folderNames = tree.folderDf["name"]
    fileNames = tree.fileDf["name"]

    if(ignoreCase):
        flags = re.IGNORECASE
    else:
        flags = None

    # Extract all matching folder entries in the name col
    if(flags != None):
        folderResults = folderNames.str.extractall(regex, flags=flags)
    else:
        folderResults = folderNames.str.extractall(regex)
    # Corresponding index value (absolute paths) of results 
    folderPaths = [result[0] for result in folderResults.index.to_list()]
    
    # Extract all matching file entries in the name col
    if(flags != None):
        fileResults = fileNames.str.extractall(regex, flags=flags)
    else:
        fileResults = fileNames.str.extractall(regex)
    # Corresponding index value (absolute paths) of results 
    filePaths = [result[0] for result in fileResults.index.to_list()]

    return (folderPaths, filePaths)

def FindSimilarSearchTerms(searchTerm:str):
    """
        Description:
        ------------
        Find the unique lemmatized synonymns of the given search term. 


        Args:
        -----  
            searchTerm : str 
                #: The term to find synonyms of 
        
        Outputs:
        --------
            synonyms : list[str] 
                #: List of synonymns found 

    """
    synsets = wn.synsets(searchTerm)
    synonyms = []

    for syn in synsets:
        synonyms.extend([lem.name() for lem in syn.lemmas()])
    
    # Remove duplicated synonymns
    synonyms = list(dict.fromkeys(synonyms)) 

    return synonyms 


# Perform a fuzzy search within the dataframe for matching items in [colName] above the threshold
def FuzzyColumnSearch(searchTerm:str, df:pd.DataFrame, colName:str, threshold=0.9):
    """
        Description:
        ------------
        Search the specified dataframe column for fuzzy matches above the threshold. 
        Uses the rapidfuzz process function to determine the similarity between strings.

        Args:
        ----- 
            searchTerm : str 
                #: The term to search for 
            df : pd.DataFrame
                #: The dataframe to search within
            colName : str
                #: The name of the dataframe column to search within
        Kwargs:
        -------
            threshold : float : default = 0.9
                #: Minimum similarity threshold
            
        Outputs:
        --------
            resultIndices : list
                #: List of dataframe indices corresponding to the search results
    """
    
    # Do nothing if passed empty search term string
    if(searchTerm == ""):
        return

    # Check if dataframe is empty
    if(not df.empty):
        # Grab the requested column
        searchItems = df[colName]
    else:
        # Return blank list if df empty
        searchItems = list()
        print("Dataframe was empty!")

    
    # Returns list of tuples (result, score(%), index) - index is dataframe index (absolute path)
    allSearchResults = process.extract(searchTerm, searchItems, limit=20)
    
    # Return the list of df indices for each item above threshold 
    resultIndices = [result[2] for result in allSearchResults if result[1] >= threshold*100]

    return resultIndices


# Aggregate results of multiple searches (same dataframe)  
def MultiTermFuzzySearch(searchTerms:list[str], df:pd.DataFrame, colName:str, uniqueOnly=False):
    """
        Description:
        ------------
        Aggregates the results of multiple fuzzy searches within the same dataframe column.
        If uniqueOnly=False the results are ordered by the number of occurences (if multiple search terms 
        matched to the same item).


        Args:
        -----  
            searchTerms : list[str] 
                #: The terms to individually search for 
            df : pd.DataFrame
                #: The dataframe to search within
            colName : str
                #: The name of the dataframe column to search within
        Kwargs:
        -------
            uniqueOnly : bool : default = False
                #: If true only count each search match once, otherwise sort matches by # of hits
            
        Outputs:
        --------
            allResults : list
                #: List of dataframe indices corresponding to the search results
            termResultsDict : dict
                #: Dict containing each search term -> matching results list
            resultTermsDict : dict
                #: Dict containing each result -> matching search terms 
    """
    allResults = list()
    # Dict storing search term -> search results
    termResultsDict = dict()
    # Dict storing search result -> terms producing that result
    resultTermsDict = defaultdict(list)
    for term in searchTerms:
        # Fuzzy search results
        results = FuzzyColumnSearch(term, df, colName)
        if(results):
            # Store the specific results for this search term
            termResultsDict[term] = results
            for result in results:
                resultTermsDict[result].append(term)
            # Remove any search results already found if requested
            if(uniqueOnly):
                for res in results:
                    if res not in allResults:
                        allResults.append(res)
            else:
                allResults.extend(results)
    if(not uniqueOnly):
        allResults = RankedResultsByOccurences(allResults, termResultsDict, resultTermsDict)

    return allResults, termResultsDict, resultTermsDict


# Use to aggregate results from searching multiple dataframes (with a common column name) 
def AggregateDataframeSearch(searchTerms:list[str], dataframes:list[pd.DataFrame], colName:str, uniqueOnly=False):
    """
        Description:
        ------------
        Used to aggregate the results of searching multiple (i.e File and Folder) dataframes that share a common column name.


        Args:
        -----  
            searchTerms : list[str] 
                #: The terms to individually search for 
            dataframmes : list[pd.DataFrame]
                #: The dataframes to search within
            colName : str
                #: The name of the dataframe column to search within
        Kwargs:
        -------
            uniqueOnly : bool : default = False
                #: If true only count each search match once, otherwise sort matches by # of hits
        
        Outputs:
        --------
            allResults : list
                #: List of dataframe indices (absolute paths) corresponding to the search results
            termResultsDict : dict
                #: Dict containing each search term -> matching results list
            resultTermsDict : dict
                #: Dict containing each result -> matching search terms 

    """
    
    allResults = list()
    termResultsDict = dict()
    resultTermsDict = defaultdict(list)
    # Collect the results from searching each dataframe
    for df in dataframes:
        results, termResults, resultTerms = MultiTermFuzzySearch(searchTerms, df, colName, uniqueOnly=uniqueOnly)
        allResults.extend(results)
        termResultsDict.update(termResults)
        resultTermsDict.update(resultTerms)
    # Rank the aggregated results if requested
    if(not uniqueOnly):
        allResults = RankedResultsByOccurences(allResults, termResultsDict, resultTermsDict)
    
    return allResults, termResultsDict, resultTermsDict

# Order search results that contain duplicates (result hit from multiple synonyms) by # of occurences
def RankedResultsByOccurences(allResults:list[str], termResultsDict:dict, resultTermDict:defaultdict[list]):
    """
        Description:
        ------------
        Used to sort the search results based on the number of matched results. Results with multiple matches (eg from several of the included search terms)
        are ranked higher.


        Args:
        -----  
            allResults : list
                #: List of dataframe indices (absolute paths) corresponding to the search results
            termResultsDict : dict
                #: Dict containing each search term -> matching results list
            resultTermsDict : dict
                #: Dict containing each result -> matching search terms 
        
        
        Outputs:
        --------
            sortedResults : list
                #: List of dataframe indices (absolute paths) corresponding to the sorted search results


    """
    # Dict containing result -> # of occurences
    occurences = Counter(allResults)
    # Sort by the # of occurences (descending) 
    sortedOccurences = occurences.most_common()
    # Separate any results that occur more than once (retains sort order)
    singleOcc = list()
    sortedMultiOcc = list()
    for occ in sortedOccurences:
        if(occ[1] > 1):
            sortedMultiOcc.append(occ[0])
        else:
            singleOcc.append(occ[0]) 
    
    # Sort the list of results by # of occurences
    sortedResults = [occ for occ in sortedMultiOcc]
    # Append single occurences 
    sortedResults.extend(singleOcc)    

    return sortedResults
